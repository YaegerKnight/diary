So what we did was for single images. It's annoying to do it multiple times so lets learn how to do it for multiple files.

searches = 'Honda CBR', 'Kawasaki Ninja','Yamaha YZF','Suzuki GSX-R'
# This basically specifies the directory in which to put the images 
path = Path('Motorbikes')

for term in searches:
  subdir = term.split()[0]
  dest = path/subdir
  dest.mkdir(exist_ok=True, parents=True)
  download_images(dest, urls=search_images_ddg(term, max_images=50))
  resize_images(path/subdir, max_size=400, dest = path/subdir)

# in their download and how to rectify it. 
failed_images = verify_images(get_image_files(path))
failed_images.map(Path.unlink)
len(failed_images)

dls = DataBlock(
    blocks=(ImageBlock, CategoryBlock),
    get_items = get_image_files,
    splitter=RandomSplitter(valid_pct=0.2, seed=42),
    get_y = parent_label,
    item_tfms = [Resize(200, method='squeeze')]
).dataloaders(path)

dls.show_batch()

learn = vision_learner(dls, resnet18, metrics=[error_rate, accuracy])
